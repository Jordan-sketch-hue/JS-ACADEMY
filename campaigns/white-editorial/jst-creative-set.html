Pexels API — Real Footage Integration
Replace the CSS mock-video background with licensed stock footage. The module fetches a portrait video, injects it at low opacity over the white background, and degrades silently to the CSS animation if the API call fails or returns no results.
pexels-video.js
// Pexels Video API — swap CSS mock-video for real footage
// Free tier: 200 req/hour · get key at pexels.com/api

const PEXELS_KEY  = 'YOUR_API_KEY';
const PEXELS_BASE = 'https://api.pexels.com/videos/search';

/**
 * Fetch a Pexels portrait video and mount it in a .mvb container.
 * Falls back silently to CSS animation on any failure.
 *
 * @param {string} query       search term
 * @param {string} containerId id of the .mvb element to inject into
 * @param {object} opts        { minDuration:20, perPage:5 }
 */
async function injectPexelsVideo(query, containerId, opts = {}) {
  const { minDuration = 20, perPage = 5 } = opts;
  const url = `${PEXELS_BASE}?query=${encodeURIComponent(query)}&orientation=portrait&per_page=${perPage}`;

  try {
    const res  = await fetch(url, { headers: { Authorization: PEXELS_KEY } });
    if (!res.ok) throw new Error(`HTTP ${res.status}`);

    const { videos } = await res.json();
    const pick = videos.find(v => v.duration >= minDuration);
    if (!pick) throw new Error('No video met minimum duration');

    const file = bestFile(pick.video_files, 720);
    mountVideo(file.link, containerId);

    return { ok: true, id: pick.id, by: pick.user.name };
  } catch (err) {
    // CSS animation already visible — no visible fallback needed
    console.warn('[JST Pexels] CSS fallback:', err.message);
    return { ok: false };
  }
}

// Pick the mp4 file closest to targetWidth
function bestFile(files, targetWidth) {
  return [...files]
    .filter(f => f.file_type === 'video/mp4' && f.width)
    .sort((a, b) => Math.abs(a.width - targetWidth) - Math.abs(b.width - targetWidth))[0]
    ?? files[0];
}

// Inject a muted looping <video> into the container at low opacity
// so the white editorial feel is preserved — footage reads as texture
function mountVideo(src, containerId) {
  const el = document.getElementById(containerId);
  if (!el) return;

  const vid = Object.assign(document.createElement('video'), {
    src, autoplay: true, loop: true, muted: true, playsInline: true,
  });
  vid.style.cssText = `
    position:absolute; inset:0;
    width:100%; height:100%; object-fit:cover;
    opacity:0; transition:opacity 1s ease;
  `;
  vid.addEventListener('canplay', () => { vid.style.opacity = '0.14'; });
  el.prepend(vid);
}
